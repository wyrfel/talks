reveal.js setup

  • fork hakimel/reveal.js
  • setup your own talks repository
  • install via bower
  • add components easily
  • re-use them in several talks

create asset directories

  • css
  • js
  • images

create bower.json

                            
bower init
                            
                        
                            
// bower.json
{
    ...
    "ignore": [
        "**/.*",
        "node_modules",
        "bower_components"
    ],
    "dependencies": {
        "reveal-js": "hakimel/reveal.js#3.1.0",
    }
}
                            
                        
                            
bower install

create Gruntfile.js

  • copy from bower_components/reveal-js/Gruntfile.js
  • delegate all tasks back to reveal.js, e.g. 
                                        
uglify: {
    gruntfile: 'bower_components/reveal-js/Gruntfile.js',
    tasks: ['uglify']
},

Bower through Grunt

npm install grunt-bower-install-simple --save-dev 
                            
grunt.initConfig({
    'bower-install-simple': {
        options: {
            production: true
        }
    }
}
grunt.loadNpmTasks('grunt-bower-install-simple');
grunt.registerTask('publish', ['bower-install-simple']);

grunt integration

npm install grunt-gh-pages --save-dev 
                            
grunt.initConfig({
    'gh-pages': {
        options: {
            push: true
        },
        src: [
            'js/**/*.js',
            'css/**/*.css',
            'images/**/*',
            'bower_components/**',
            'index.html',
            '.../**/*'
        ]
    }
}
grunt.loadNpmTasks('grunt-gh-pages');
grunt.registerTask('publish', ['bower-install-simple', 'gh-pages']);
grunt publish
